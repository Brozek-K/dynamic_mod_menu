country_event = {
    id = dynamic_mod_menu_utilities.1
    title = "dmm_options.1.title"
    desc = "dmm_options.1.desc"
    is_triggered_only = yes
    diplomatic = yes
    custom_gui = "dynamic_mod_menu_utilities_ui"

    trigger = {
        NOT = {
            has_global_flag = dynamic_mod_menu_utilities_opened
        }
    }

    immediate = {
        hidden_effect = {
            if = {
                limit = {
                    NOT = {
                        has_global_flag = dynamic_mod_menu_utilities_installed
                    }
                }
                set_global_flag = dynamic_mod_menu_utilities_installed
            }
            set_global_flag = dynamic_mod_menu_utilities_opened
            set_variable = {
                which = dynamic_mod_menu_utilities_processed_id
                value = 0
            }
            set_variable = {
                which = dynamic_mod_menu_utilities_last_processed_id
                value = 0
            }
            set_variable = {
                which = dynamic_mod_menu_utilities_last_id
                value = 0
            }
            set_variable = {
                which = dynamic_mod_menu_utilities_first_id
                value = 0
            }
        }
        country_event = {
            id = dynamic_mod_menu_utilities.2
        }
        country_event = {
            id = dynamic_mod_menu_utilities.3
        }
        country_event = {
            id = dynamic_mod_menu_utilities_pagination.4
        }
        country_event = {
            id = dynamic_mod_menu_utilities_pagination.1
        }
    }

    option = {
        name = "dmm_options.1.close"

        hidden_effect = {
            remove_global_flag = dynamic_mod_menu_utilities_opened
        }
    }
}
